class test {

	public void main(char[][] args) {
		print(this.gcd(2,14));
		print(this.gcd(3,15));
		print(this.gcd(99,121));
	}

	public int gcd(int x, int y){
		int a = x;
		int b = y;
  		while (a != b) {
    		if (a > b)
    			a = a - b;
    		else 
    			b = b - a;
  			}
  		return a;
	}
}
